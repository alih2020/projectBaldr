package formatif21;

import java.util.Scanner;

/**

*

* @author usager

*/

public class formatif21 {

/**

* @param args the command line arguments Cette méthode main déclare toutes

* les valeurs essentielles qui seront utiliser tout au long du travail. De

* plus, ce méthode main comporte aussi plusieurs méthodes qui seront

* évidemment entrer à partir de cette méthode là.

*/

public static void main(String[] args) {

// Déclaration de la méthode pour l'ouverture de la ligne sportive

Introduction();

int nombrelut;

// Déclaration de la méthode pour déterminer le nombre d'équipes

nombrelut = DeterminerEquipe();

String[] nom = new String[nombrelut];

// Déclaration des tableaux qui seront utiliser par la suite

int[] PJ = new int[nombrelut];

int[] victoire = new int[nombrelut];

int[] defaite = new int[nombrelut];

int[] pointstotal = new int[nombrelut];

NommerEquipe(nom);

AffichageTableauEquipe(nom, PJ, victoire, defaite, pointstotal);

MenuEntree(nom, PJ, victoire, defaite, pointstotal);

}

/**

* Cette méthode introduira la ligne sportive tout simplement

*/

private static void Introduction() {

System.out.println("Bienvenu dans la ligue sportive où 3 équipes");

System.out.println("ou plus compétitionneront entre eux. L'équipe avec");

System.out.println("le plus de points sera le champion de la ligue ");

System.out.println("sportive et ainsi remportera une médaille d'or!");

System.out.println(" Faite l'analyse des résulats obtenus: ");

System.out.println("-------------------------------------------------");

}

/**

* Cette méthode demandera le nombre d'équipe et dira qu'il doit avoir au

* moins 3 équipes. Le tableau prendra la taille du nombre qu'on aurait

* entrer

*

* @return le nombre d'équipe qui est entrer

*/

private static int DeterminerEquipe() {

Scanner r = new Scanner(System.in);

int nombreEquipe;

do {

System.out.println("Combien y a-t-il d'équipes dans la ligue?");

nombreEquipe = r.nextInt();

} while (nombreEquipe < 3);

return nombreEquipe;

}

/**

* Cette méthode permettra d'afficher les noms des équipes,elle affichera

* autant d'équipes que nous avions définie dans la méthode préédente. Les

* noms des équipes ne peuvent être similaires et ne peuvent être vides. Et

* si cela est le cas,l'ordinateur doit afficher un message d'erreur

*

* @param nom

*/

private static void NommerEquipe(String[] tab) {

Scanner r = new Scanner(System.in);

// Tant que toutes les nom des équipes n'ont pas été entrées

// lire un nom d'équipe

// Vérifier que le nom d'équipe n'Existe pas déjà

// Resaissir si le nom d'équipe est vide

for (int i = 0; i < tab.length; i++) {

System.out.println("Entrez un nom pour une équipe: ");

tab[i] = r.nextLine();

if (tab[i].length() == 0) {

System.out.println("Un nom d'équipe ne peut rester vide");

i--;

}

for (int j = 0; j < i; j++) {

if (tab[j].equals(tab[i])) {

System.out.println("Cette équipe existe déja!!");

System.out.println("");

i--;

}

}

}

}

/**

* Affichage du tableau où chacune de ces paramètres sont à 0 par default

* Dans d'autres mots, Chaque colonne ou case des tableaux seront à 0. Les

*

* @param = paramètres ne sont que des tableaux.

*

* @param nom

* @param PJ

* @param victoire

* @param defaite

* @param pointstotal

*/

private static void AffichageTableauEquipe(String[] nom, int[] PJ,

int[] victoire, int[] defaite, int[] pointstotal) {

System.out.println("==================================================");

System.out.println("Nom d'équipe" + "\t" + "PJ" + "\t" + "victoire" + "\t" + "defaite" + "\t" + "points");

System.out.println("---------------------------------------------------");

for (int i = 0; i < pointstotal.length; i++) {

System.out.println(nom[i] + "\t" + "\t" + PJ[i] + "\t" + victoire[i] + "\t" + "\t" + defaite[i] + "\t" + pointstotal[i]);

}

}

/**

* Tant que l'utilisateur entre autre que 1 ou 2 comme choix, l'ordinateur

* demande une autre saisie en mettant une option acceptable Si

* l'utilisateur entre 1, on entre dans une autre méthode,appelé

* ClasserTeam. Si l'utilisateur entre 2 comme choix, alors on aurait à

* quitter la ligue sportive sur le champs.

*

* @param nom

* @param PJ

* @param victoire

* @param defaite

* @param pointstotal

*/

private static void MenuEntree(String[] tab, int[] PJ, int[] victoire,

int[] defaite, int[] pointstotal) {

Scanner r = new Scanner(System.in);

int choix;

do {

System.out.println("===============================================");

System.out.println("");

System.out.println("1) Entrez un résultat");

System.out.println("2) Quitter");

System.out.println("-----------------------------------------------");

do {

System.out.println("Entrez votre choix: ");

choix = r.nextInt();

switch (choix) {

case 1:

// Si choix 1 | On entre dans une méthode différente

ClasserTeam(tab, PJ, victoire, defaite, pointstotal);

break;

case 2:

break;

default:

System.out.println("Entrez une option valable!!");

}

} while (choix != 1 && choix != 2);

// Si choix = 2 | l'utilisateur quitte la ligue sportive sur le champs

} while (choix != 2);

System.out.println("");

System.out.println("Merci d'avoir assister à la ligue sportive!");

}

/**

* Cette méthode s'applique uniquement si nous avons entrer 1 comme choix

* dans la méthode précédente,dans la méthode MenuEntree. C'est l'option 1

* qui a permis d'entrer dans cette méthode là. Cette méthode doit classer

* les équipes d'Abord et demander par la suite l'équipe gagnante ainsi que

* l'équipe perdante

*

* @param nom

* @param PJ

* @param victoire

* @param defaite

* @param pointstotal

*/

private static void ClasserTeam(String[] nom, int[] PJ, int[] victoire, int[] defaite, int[] pointstotal) {

Scanner r = new Scanner(System.in);

int equipegagnante;

int equipeperdante;

System.out.println("Vous avez choisi d'entrer un résultat ");

System.out.println("Les équipes sont les suivantes:");

for (int i = 0; i < nom.length; i++) {

System.out.println(i + 1 + ")" + " " + nom[i]);

}

System.out.println("Identifier l'équipe gagnante: ");

equipegagnante = r.nextInt();

// Calcule pour entrer le donnée dans le tableau victoire

victoire[equipegagnante - 1] += 1;

do {

System.out.println("Identifier l'équipe perdante: ");

equipeperdante = r.nextInt();

//Condition pour l'affichage du message et pour des buts estétiques

if (equipegagnante == equipeperdante) {

System.out.println("L'équipe gagnante ne peut être l'équipe" + " perdante");

System.out.println("");

}

} while (equipegagnante == equipeperdante);

// Calcule pour entrer le donnée dans le tableau défaite

defaite[equipeperdante - 1] += 1;

AugmenterlesPoints(PJ, victoire, defaite, pointstotal);

AffichageTableauEquipe(nom, PJ, victoire, defaite, pointstotal);

}

/*

* Cette méthode permettra de modifier les données du tableau par rapport à

* ce que nous entrions. Elle calculera les données comme l'ensemble de

* victoires et de défaites équivaut au partis jouées et la multiplication

* des victoires par 2 équivaut au point totals entre les 2 équipes qui sont

* en compétition, le tableau modifié se reaffichera.

*

* @param PJ

* @param victoire

* @param defaite

* @param pointstotal

*/

private static void AugmenterlesPoints(int[] PJ, int[] victoire,

int[] defaite, int[] pointstotal) {

for (int i = 0; i < PJ.length; i++) {

PJ[i] = victoire[i] + defaite[i];

}

for (int i = 0; i < pointstotal.length; i++) {

pointstotal[i] = victoire[i] * 2;

}

}

}
